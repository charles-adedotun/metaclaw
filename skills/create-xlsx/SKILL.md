## Step 3: Template

```python
import openpyxl
from openpyxl.styles import Font, PatternFill, Alignment, Border, Side, numbers
from openpyxl.utils import get_column_letter
from datetime import date

# --- THEME (adapt per design-system) ---
T = {
    "primary":   "1B2A4A",
    "secondary": "2E5090",
    "accent":    "C8A97E",
    "bg_alt":    "EEF1F5",
    "text_body": "2D3436",
    "text_muted": "888888",
    "border":    "D5DAE0",
}

wb = openpyxl.Workbook()
ws = wb.active
ws.title = "Report"  # ALWAYS use descriptive sheet names

# --- Styles ---
header_font = Font(name="Calibri", size=12, bold=True, color="FFFFFF")
header_fill = PatternFill(start_color=T["secondary"], end_color=T["secondary"], fill_type="solid")
body_font = Font(name="Calibri", size=11, color=T["text_body"])
title_font = Font(name="Calibri", size=16, bold=True, color=T["primary"])
subtitle_font = Font(name="Calibri", size=10, italic=True, color=T["text_muted"])
alt_fill = PatternFill(start_color=T["bg_alt"], end_color=T["bg_alt"], fill_type="solid")
accent_fill = PatternFill(start_color=T["accent"], end_color=T["accent"], fill_type="solid")
thin_border = Border(
    left=Side(style="thin", color=T["border"]),
    right=Side(style="thin", color=T["border"]),
    top=Side(style="thin", color=T["border"]),
    bottom=Side(style="thin", color=T["border"]),
)

# --- Title Row ---
ws.merge_cells("A1:D1")  # Adjust to column count
ws["A1"] = "Report Title"
ws["A1"].font = title_font
ws["A1"].alignment = Alignment(horizontal="left", vertical="center")
ws.row_dimensions[1].height = 32

# --- Subtitle ---
ws["A2"] = f"Generated by Andy  •  {date.today().strftime('%B %d, %Y')}"
ws["A2"].font = subtitle_font
ws.row_dimensions[2].height = 22

# --- Header Row (row 3) ---
headers = ["Column A", "Column B", "Column C", "Column D"]
for col_idx, header in enumerate(headers, 1):
    cell = ws.cell(row=3, column=col_idx, value=header)
    cell.font = header_font
    cell.fill = header_fill
    cell.alignment = Alignment(horizontal="center", vertical="center")
    cell.border = thin_border
ws.row_dimensions[3].height = 30

# --- Data Rows ---
data = [
    ["Item 1", 1500.00, 0.125, "Active"],
    ["Item 2", 2300.50, 0.087, "Pending"],
    ["Item 3", 980.75, 0.203, "Active"],
]
for row_idx, row_data in enumerate(data, 4):
    for col_idx, value in enumerate(row_data, 1):
        cell = ws.cell(row=row_idx, column=col_idx, value=value)
        cell.font = body_font
        cell.border = thin_border
        cell.alignment = Alignment(vertical="center")
        # Alternating rows
        if (row_idx - 4) % 2 == 1:
            cell.fill = alt_fill
        # Smart number formatting
        if isinstance(value, float) and col_idx == 2:
            cell.number_format = '#,##0.00'
        elif isinstance(value, float) and col_idx == 3:
            cell.number_format = '0.0%'

# --- Auto-fit columns ---
for col_idx in range(1, len(headers) + 1):
    max_len = max(
        len(str(ws.cell(row=r, column=col_idx).value or ""))
        for r in range(3, 4 + len(data))
    )
    ws.column_dimensions[get_column_letter(col_idx)].width = max(max_len + 4, 14)

# --- Freeze header ---
ws.freeze_panes = "A4"

wb.save("/workspace/group/output.xlsx")
```

## Step 4: Advanced Patterns

### Summary Row (totals, averages)
```python
total_row = 4 + len(data)
ws.cell(row=total_row, column=1, value="TOTAL").font = Font(name="Calibri", size=11, bold=True, color=T["primary"])
ws.cell(row=total_row, column=2, value=f"=SUM(B4:B{total_row-1})").font = Font(name="Calibri", size=11, bold=True, color=T["primary"])
ws.cell(row=total_row, column=2).number_format = '#,##0.00'
for c in range(1, len(headers) + 1):
    ws.cell(row=total_row, column=c).border = Border(
        top=Side(style="medium", color=T["primary"]),
        bottom=Side(style="thin", color=T["border"]),
        left=Side(style="thin", color=T["border"]),
```
